\section{理论推导}
\begin{frame}
  \frametitle{理论推导}
  \begin{equation}
    \label{eq:01}
    y_1=f(Image,Points)
  \end{equation}
  \begin{equation}
    \label{eq:02}
    y_2=f(g(Image,Points))
  \end{equation}
  \begin{equation}
    \label{eq:03}
  ||y-f(g(I,P))|| \leq ||y-f(I,P)||    
  \end{equation}

\end{frame}

\begin{frame}
  \frametitle{理论推导}
  设我们的输入为$x_1=(Image,Points)$,
  $g(x_1)=x_2$,且$D(x_1)>D(x_2)$.
  然后一般的，
  如果$f(x)$有任意价的
  我们有$f(x)=x_0+f^{'}(x_0)x+f^{''}(x_0)x^2+f^{3}(x_0)x^3+\dots$

\end{frame}
\begin{frame}
  \frametitle{推导}
    \begin{equation}
    \label{eq:05}
    \begin{aligned}[t]
      D(f(x)) & =D(x_0+f^{'}(x_0)x+f^{''}(x_0)x^2+f^{3}(x_0)x^3+\dots),\\
      &= f^{'}(x_0)D(x)+f^{''}(x_0)D(x^2)+f^3(x_0)D(x^3)+\dots,\\
      &=(f^{'}(x_0)+a_2f^{''}(x_0)+a_3f^3(x_0)+\dots)\sigma^2,\\
      &=K\sigma^2
    \end{aligned}
  \end{equation}
  当$k$为奇数时，
  \begin{equation}
    \label{eq:06}
    D(x^k)=(2k-1)!! \sigma^2
  \end{equation}
  当$k$为偶数时，
    \begin{equation}
    \label{eq:07}
    D(x^k)=((2k-1)!!-(k-1)!!) \sigma^2
  \end{equation}
\end{frame}
